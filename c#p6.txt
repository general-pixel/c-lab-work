6.write a program to input age of a person and display message as follows-if age <12 print you are a kid, if age between 12 to 17 print you are teenager and if age between 18 to 60 print your adult and if age>60 print you are senior citizen

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter your age: ");
        int age = int.Parse(Console.ReadLine());

        if (age < 12)
        {
            Console.WriteLine("You are a kid");
        }
        else if (age >= 12 && age <= 17)
        {
            Console.WriteLine("You are a teenager");
        }
        else if (age >= 18 && age <= 60)
        {
            Console.WriteLine("You are an adult");
        }
        else
        {
            Console.WriteLine("You are a senior citizen");
        }
    }
}
